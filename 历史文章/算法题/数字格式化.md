# 数字格式化

## 1. 将1234567890 --> 1,234,567,890

```JavaScript
function format(num) {
  if (typeof num !== "number") {
    // 非数字直接返回
    return;
  }
  const str = num.toString();
  let len = str.length;
  let rightIndex = 0;
  let arr = [];
  let temp = "";
  while (len - rightIndex > 0) {
    rightIndex;
    temp = `${str[len - rightIndex - 1]}${temp}`;
    rightIndex++;
    if (rightIndex % 3 === 0) {
      arr.unshift(temp);
      temp = "";
    }
  }
  if (temp.length > 0) {
    arr.unshift(temp);
  }
  return arr.join(",");
}

console.log(format(1234567890)); // 输出1,234,567,890

```

function compareVersion(a, b) {
    if (a === b) return 0;
    const aSplit = a.split('.');
    const bSplit = b.split('.');
    while(aSplit.length || bSplit.length) {
        const aCurrent = aSplit.shift()
        const bCurrent = bSplit.shift()
        if (aCurrent === undefined) {
            return -1;
        }
        if (bCurrent === undefined) {
            return 1;
        }
        const aNumber = parseInt(aCurrent), bNumber = parseInt(bCurrent);
        if (aNumber !== bNumber) {
            return aNumber - bNumber;
        }
        // 否则一样大就继续比
    }
}

function sortVersions(versionArray) {
    return versionArray.sort(compareVersion)
}

var versions=['1.45.0','1.5', '1.5.1','6','3.3.3.3.3.3.3']
console.log(sortVersions(versions))
